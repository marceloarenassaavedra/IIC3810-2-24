\begin{enumerate}
\item Defina el siguiente lenguaje:
\begin{multline*}
\text{SUBSET-SUM} \ = \ 
\{ (S,a) \mid S \cup \{a\} \text{ es un conjunto de número naturales y}\\
\text{existe } S' \subseteq S \text{ tal que }
\sum_{b \in S'} b = a\}
\end{multline*}
Demuestre que $\tcnf \rp \text{SUBSET-SUM}$.

\item Modifique $\text{SUBSET-SUM}$ para obtener un problem
  $\Sigma_2^P$-completo. No es necesario que dé una demostración formal de que la versión
  modificada es $\Sigma_2^P$-completo, pero sí debe explicar por qué se cumple esta propiedad.
\end{enumerate}

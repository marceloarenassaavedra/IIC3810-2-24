\begin{enumerate}
\item Sea $f : \Sigma^* \to \mathbb{N}$ una función que admite un
  FPRAS $\cA$. Utilizando la idea de hacer varias llamada a $\cA$ y
  tomar la mediana de los resultados, demuestre que para cada $\ell
  \geq 1$, existe un FPRAS $\cB$ para $f$ tal que para cada $x \in
  \Sigma^*$ y $\varepsilon \in (0,1)$:
  \begin{eqnarray*}
    \pr(|\cB(x, \varepsilon) - f(x)| \leq \varepsilon \cdot f(x)) &\geq& 1 - \bigg(\frac{3}{4}\bigg)^\ell.
  \end{eqnarray*}


\item Demuestre que si las funciones $f : \Sigma^* \to \mathbb{N}$ y
  $g : \Sigma^* \to \mathbb{N}$ admiten FPRAS, entonces $f + g$ y $f
  \cdot g$ también admiten FPRAS.

\item Para cada $n \in \mathbb{N}$, sea $f_n : \Sigma^* \to
  \mathbb{N}$ una función que admite un FPRAS $\cA_n$, donde el número
  de pasos ejecutados por $\cA_n(x, \varepsilon)$ está acotado
  superiormente por $p_n(|x|, \frac{1}{\varepsilon})$ para un
  polinomio fijo $p_n$. Además, suponga que existe un polinomio
  $p(u, v)$ que acota superiormente a $p_n(u, v)$ para cada $n \in \mathbb{N}$.
  Sea $f : \Sigma^* \to \mathbb{N}$ una función definida de la
  siguiente forma para cada $x \in \Sigma^*$:
  \begin{eqnarray*}
    f(x) &=& \sum_{i=0}^{|x|} f_i(x).
  \end{eqnarray*}
  Demuestre que $f$ admite un FPRAS.
  

\end{enumerate}
